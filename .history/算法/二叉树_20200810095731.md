function binaryTree(data) {
  const { value, left, right } = data;

// 二叉树的层次遍历
function binaryTreeGradation(data, queue = []) {
  const { value, left, right, root } = data;
  // queue.push(value);
  debugger;
  if (left && left.value) {
    queue.push(left);
  }
  if (right && right.value) {
    queue.push(right);
  }
  const popValue = queue.shift(); // 通过队列 先进先出
  root && console.log({ popValue: value });
  popValue && console.log({ popValue: popValue.value });
  popValue && binaryTreeGradation(popValue, queue);
}

// 前序遍历
function traversePre(data) {
  forEach(data, (dataItem) => {
    const { id, children } = dataItem;
    console.log({ id });
    if (!isUndefined(children)) {
      traversePre(children);
    }
  });
}

// 后序遍历
function traverseAfter(data) {
  forEach(data, (dataItem) => {
    const { id, children } = dataItem;
    if (!isUndefined(children)) {
      traverseAfter(children);
    }
    console.log({ id });
  });
}

// 中序遍历
// 但是有问题 因为用数组 他不知道是左边还是右边的节点 都会把节点当作是第一个left 节点 但是其实比如6 那个节点是右节点会被误认为是左节点 所以需要在数据结构里加上标识是左节点还是右节点
function traverseInorder(data) {
  const { id, children, type } = data;
  if (children && children[0]) {
    if (type !== "right") {
      // 加上type后就可以正确遍历了
      traverseInorder(children[0]);
    }
  }
  id && console.log({ id });
  if (children && children[1]) {
    traverseInorder(children[1]);
  }
}

function iterator(arr) {
  let i = 0;
  return {
    next: function () {
      const done = i >= arr.length;
      const value = done ? undefined : arr[i++];
      return {
        done,
        value,
      };
    },
  };
}
